18  /**
19   * Subclass of {@link java.lang.RuntimeException} used to signal that there
20   * was an {@link java.io.IOException} thrown by underlying {@link java.io.Writer}
21   */
22  public class RuntimeIOException extends RuntimeException
23  {
24      public RuntimeIOException()
25      {
26          super();
27      }
28  
29      public RuntimeIOException(String message)
30      {
31          super(message);
32      }
33  
34      public RuntimeIOException(Throwable cause)
35      {
36          super(cause);
37      }
38  
39      public RuntimeIOException(String message, Throwable cause)
40      {
41          super(message,cause);
42      }
43  }
