15  package org.mortbay.io;
16  
17  import org.mortbay.util.StringUtil;
18  
19  /**
20   * Portability class containing methods not available on all JVMs (specifically SuperWaba).
21   * @author gregw
22   *
23   */
24  public class Portable
25  {
26      public static final String ALL_INTERFACES="0.0.0.0";
27      
28      public static void arraycopy(byte[] src, int srcOffset, byte[] dst, int dstOffset, int length)
29      {
30          System.arraycopy(src, srcOffset, dst, dstOffset, length);
31      }
32  
33      public static void throwNotSupported()
34      {
35          throw new RuntimeException("Not Supported");
36      }
37  
38      public static byte[] getBytes(String s)
39      {
40          try
41          {
42              return s.getBytes(StringUtil.__ISO_8859_1);
43          }
44          catch (Exception e)
45          {
46              return s.getBytes();
47          }
48      }
49  }
