15  package org.mortbay.jetty.client.webdav;
16  
17  import java.io.IOException;
18  
19  import javax.servlet.http.HttpServletResponse;
20  
21  import org.mortbay.io.Buffer;
22  import org.mortbay.jetty.client.HttpExchange;
23  import org.mortbay.log.Log;
24  
25  
26  public class PropfindExchange extends HttpExchange
27  {
28      boolean _propertyExists = false;
29  
30      /* ------------------------------------------------------------ */
31      protected void onResponseStatus(Buffer version, int status, Buffer reason) throws IOException
32      {
33          if ( status == HttpServletResponse.SC_OK )
34          {
35              Log.debug( "PropfindExchange:Status: Exists" );
36              _propertyExists = true;
37          }
38          else
39          {
40              Log.debug( "PropfindExchange:Status: Not Exists" );
41          }
42  
43          super.onResponseStatus(version, status, reason);
44      }
45  
46      public boolean exists()
47      {
48          return _propertyExists;
49      }
50  }
