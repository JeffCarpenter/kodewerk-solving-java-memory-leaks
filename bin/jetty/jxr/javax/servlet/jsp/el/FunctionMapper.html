26  /**
27   * <p>The interface to a map between EL function names and methods.</p>
28   *
29   * <p>Classes implementing this interface may, for instance, consult tag library
30   * information to resolve the map. </p>
31   *
32   * @since 2.0
33   */
34  public interface FunctionMapper {
35    /**
36     * Resolves the specified local name and prefix into a Java.lang.Method.
37     * Returns null if the prefix and local name are not found.
38     * 
39     * @param prefix the prefix of the function, or "" if no prefix.
40     * @param localName the short name of the function
41     * @return the result of the method mapping.  Null means no entry found.
42     **/
43    public java.lang.reflect.Method resolveFunction(String prefix, 
44        String localName);
45  }
